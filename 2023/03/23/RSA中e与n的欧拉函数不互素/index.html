    def CRT(a, n):
    sum = 0
    N = reduce(lambda x, y: x * y, n)
    for n_i, a_i in zip(n, a):  # zip()
        N_i = N // n_i 
        sum += a_i * N_i * invert(N_i, n_i)
    return sum % N
