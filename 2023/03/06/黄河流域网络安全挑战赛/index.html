def Cyber_key(LEN):
    Key = [[] for i in range(row)]
    for x in range(row):
        for i in range(LEN):
            Key[x].append(random.randint(0, 2023))
    return Key


def Punk_enc(Key, msg):
    out = []
    for l in Key:
        s = 0
        for x, y in zip(l, msg):
            s += (x * y) % P
            s %= P
        out.append(s)
    return out

    def CRT(a, n):
    sum = 0
    N = reduce(lambda x, y: x * y, n)  # ni 的乘积,N=n1*n2*n3

    for n_i, a_i in zip(n, a):  # zip()将对象打包成元组
        N_i = N // n_i  # Mi=M/ni
        sum += a_i * N_i * invert(N_i, n_i)
    return sum % N
