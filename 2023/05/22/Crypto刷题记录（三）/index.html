    n = [...]

for i in range(len(n)):
    for j in range(i+1,len(n)):
        if gcd(n[i],n[j]) != 1:
            p = gcd(n[i],n[j])
            q = n[j] // p

            phi = (p-1)*(q-1)
            d = invert(e, phi)
            privkey = RSA.construct((int(p * q), int(e), int(d), int(p), int(q)))
            key = PKCS1_OAEP.new(privkey)
            flag = key.decrypt(long_to_bytes(c[j]))
            print(flag)
