def AMM(o, r, q):
    start = time.time()

        p = g(random.randint(1, q))
    print('[+] Find p:{}'.format(p))
    t = 0
    s = q - 1
    while s % r == 0:
        t += 1
        s = s // r
    print('[+] Find s:{}, t:{}'.format(s, t))

        k += 1

    a = p ^ (r**(t-1) * s)
    b = o ^ (r*alp - 1)
