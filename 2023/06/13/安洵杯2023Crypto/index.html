    def continued_fraction(dn,n):
    res = []
    while dn % n:
        res.append(dn//n)
        dn, n = n, dn % n
    res.append(dn//n)
    print(res)
    return res

    def Convergence_function(continued):
    res =[]
    for i in range(1,len(continued)+1):
        tmp = 1
        conver = continued[:i][::-1]
        tmp = conver[0]
        for j in conver[1:]:
            tmp = j + 1/tmp
        res.append(tmp)
    return res
